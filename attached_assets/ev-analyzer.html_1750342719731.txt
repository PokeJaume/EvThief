    <script>
        let pokemonData = {};
        let filteredData = {};
        let showOnlyPopular = false;
        let currentSort = 'usage';

        document.getElementById('jsonFile').addEventListener('change', handleFileSelect);
        document.getElementById('searchPokemon').addEventListener('input', filterResults);
        document.getElementById('searchEVs').addEventListener('input', filterResults);

        function handleFileSelect(event) {
            const file = event.target.files[0];
            if (!file) return;

            document.getElementById('fileName').innerHTML = `<strong>📄 ${file.name}</strong>`;
            document.getElementById('loading').style.display = 'block';
            document.getElementById('error').style.display = 'none';

            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const data = JSON.parse(e.target.result);
                    processSmogonData(data);
                } catch (error) {
                    showError('Error al leer el archivo JSON. Asegúrate de que sea un archivo válido de Smogon chaos.');
                    console.error('Error parsing JSON:', error);
                }
            };
            reader.readAsText(file);
        }

        function processSmogonData(data) {
            pokemonData = {};
            let totalSpreads = 0;
            
            // Debug: mostrar estructura del JSON
            console.log('Estructura del JSON:', Object.keys(data).slice(0, 5));
            console.log('Primera entrada:', Object.entries(data)[0]);
            
            // Estructura específica de Smogon chaos con data.data
            const dataToProcess = data.data || data;
            
            // Procesar cada Pokémon en el JSON
            for (const [pokemonName, pokemonInfo] of Object.entries(dataToProcess)) {
                console.log(`Procesando ${pokemonName}:`, Object.keys(pokemonInfo));
                
                // Estructura específica del archivo que subiste
                let spreadsData = null;
                let totalCount = 0;
                
                if (pokemonInfo.Spreads) {
                    // Estructura del archivo chaos actual: pokemonInfo.Spreads
                    spreadsData = pokemonInfo.Spreads;
                    totalCount = pokemonInfo["Raw count"] || 0;
                } else if (pokemonInfo.Raw && pokemonInfo.Raw.spreads) {
                    // Estructura típica chaos alternativa
                    spreadsData = pokemonInfo.Raw.spreads;
                    totalCount = pokemonInfo.Raw.count;
                } else if (pokemonInfo.spreads) {
                    // Estructura alternativa lowercase
                    spreadsData = pokemonInfo.spreads;
                    totalCount = pokemonInfo.count || pokemonInfo["raw count"] || 0;
                }
                
                if (!spreadsData) {
                    console.log(`No se encontraron spreads para ${pokemonName}`);
                    continue;
                }

                const spreads = [];
                
                // Procesar cada spread
                for (const [spreadString, usage] of Object.entries(spreadsData)) {
                    // Verificar si es un spread válido (formato Nature:HP/Atk/Def/SpA/SpD/Spe)
                    if (!spreadString.includes(':') || !spreadString.includes('/')) continue;
                    
                    const evs = parseEVSpread(spreadString);
                    const usageNum = parseFloat(usage) || 0; // Puede ser decimal
                    const percentage = totalCount > 0 ? ((usageNum / totalCount) * 100).toFixed(2) : '0.00';
                    
                    spreads.push({
                        spread: spreadString,
                        evs: evs,
                        usage: usageNum,
                        percentage: parseFloat(percentage),
                        formatted: formatEVSpread(evs)
                    });
                    totalSpreads++;
                }

                if (spreads.length === 0) {
                    console.log(`No spreads válidos encontrados para ${pokemonName}`);
                    continue;
                }

                // Ordenar spreads por uso descendente
                spreads.sort((a, b) => b.usage - a.usage);

                pokemonData[pokemonName] = {
                    name: pokemonName,
                    spreads: spreads,
                    totalUsage: totalCount
                };
                
                console.log(`${pokemonName}: ${spreads.length} spreads procesados (total count: ${totalCount})`);
            }

            console.log('Pokémon procesados:', Object.keys(pokemonData).length);
            console.log('Total spreads:', totalSpreads);
            
            filteredData = { ...pokemonData };
            updateStatsSummary(totalSpreads);
            displayResults();
            document.getElementById('loading').style.display = 'none';
            document.getElementById('statsummary').style.display = 'block';
            
            // Mostrar mensaje si no se encontraron datos
            if (Object.keys(pokemonData).length === 0) {
                showError('No se pudieron procesar los datos. Revisa la consola del navegador (F12) para más detalles sobre la estructura del archivo.');
            }
        }

        function parseEVSpread(spreadString) {
            // Formato: "Nature:HP/Atk/Def/SpA/SpD/Spe" ej: "Impish:252/36/156/0/60/4"
            const parts = spreadString.split(':');
            if (parts.length !== 2) return { HP: 0, Atk: 0, Def: 0, SpA: 0, SpD: 0, Spe: 0 };
            
            const nature = parts[0];
            const values = parts[1].split('/');
            
            if (values.length !== 6) return { HP: 0, Atk: 0, Def: 0, SpA: 0, SpD: 0, Spe: 0 };
            
            return {
                nature: nature,
                HP: parseInt(values[0]) || 0,
                Atk: parseInt(values[1]) || 0,
                Def: parseInt(values[2]) || 0,
                SpA: parseInt(values[3]) || 0,
                SpD: parseInt(values[4]) || 0,
                Spe: parseInt(values[5]) || 0
            };
        }

        function formatEVSpread(evs) {
            const statNames = { HP: 'HP', Atk: 'Ataque', Def: 'Defensa', SpA: 'At.Esp', SpD: 'Def.Esp', Spe: 'Velocidad' };
            const nonZero = Object.entries(evs).filter(([stat, value]) => stat !== 'nature' && value > 0);
            
            if (nonZero.length === 0) return evs.nature ? `${evs.nature}: Sin EVs` : 'Sin EVs';
            
            const natureText = evs.nature ? `${evs.nature}: ` : '';
            return natureText + nonZero.map(([stat, value]) => `${statNames[stat]}: ${value}`).join(', ');
        }

        function updateStatsSummary(totalSpreads) {
            const totalPokemon = Object.keys(pokemonData).length;
            const avgSpreads = totalPokemon > 0 ? (totalSpreads / totalPokemon).toFixed(1) : 0;
            
            document.getElementById('totalPokemon').textContent = totalPokemon;
            document.getElementById('totalSpreads').textContent = totalSpreads;
            document.getElementById('avgSpreads').textContent = avgSpreads;
        }

        function displayResults() {
            const resultsDiv = document.getElementById('results');
            resultsDiv.innerHTML = '';
            
            if (Object.keys(filteredData).length === 0) {
                resultsDiv.innerHTML = '<div class="error-message">No se encontraron resultados con los filtros actuales.</div>';
                resultsDiv.style.display = 'block';
                return;
            }

            let sortedPokemon = Object.values(filteredData);
            
            // Aplicar ordenamiento
            if (currentSort === 'usage') {
                sortedPokemon.sort((a, b) => b.totalUsage - a.totalUsage);
            } else if (currentSort === 'name') {
                sortedPokemon.sort((a, b) => a.name.localeCompare(b.name));
            }

            sortedPokemon.forEach(pokemon => {
                const pokemonDiv = document.createElement('div');
                pokemonDiv.className = 'pokemon-card';
                
                const spreadsHtml = pokemon.spreads.map(spread => `
                    <div class="ev-spread">
                        <div class="ev-values">
                            <div class="ev-stat" style="background: #e74c3c;">${spread.evs.nature || 'Neutral'}</div>
                            <div class="ev-stat">HP: ${spread.evs.HP}</div>
                            <div class="ev-stat">Atk: ${spread.evs.Atk}</div>
                            <div class="ev-stat">Def: ${spread.evs.Def}</div>
                            <div class="ev-stat">SpA: ${spread.evs.SpA}</div>
                            <div class="ev-stat">SpD: ${spread.evs.SpD}</div>
                            <div class="ev-stat">Spe: ${spread.evs.Spe}</div>
                        </div>
                        <div class="usage-percent">${spread.percentage}%</div>
                    </div>
                `).join('');

                pokemonDiv.innerHTML = `
                    <div class="pokemon-name">
                        🎯 ${pokemon.name} (${pokemon.spreads.length} spreads únicos)
                    </div>
                    ${spreadsHtml}
                `;
                
                resultsDiv.appendChild(pokemonDiv);
            });
            
            resultsDiv.style.display = 'block';
        }

        function filterResults() {
            const pokemonFilter = document.getElementById('searchPokemon').value.toLowerCase();
            const evFilter = document.getElementById('searchEVs').value.toLowerCase();
            
            filteredData = {};
            
            for (const [name, pokemon] of Object.entries(pokemonData)) {
                // Filtrar por nombre de Pokémon
                if (pokemonFilter && !name.toLowerCase().includes(pokemonFilter)) {
                    continue;
                }
                
                let filteredSpreads = [...pokemon.spreads];
                
                // Filtrar por EVs específicos
                if (evFilter) {
                    filteredSpreads = filteredSpreads.filter(spread => {
                        const spreadText = spread.spread.toLowerCase() + ' ' + spread.formatted.toLowerCase();
                        return spreadText.includes(evFilter) || 
                               Object.values(spread.evs).join('/').includes(evFilter.replace(/\D/g, ''));
                    });
                }
                
                // Filtrar por popularidad
                if (showOnlyPopular) {
                    filteredSpreads = filteredSpreads.filter(spread => spread.percentage >= 5);
                }
                
                if (filteredSpreads.length > 0) {
                    filteredData[name] = {
                        ...pokemon,
                        spreads: filteredSpreads
                    };
                }
            }
            
            displayResults();
        }

        function sortBy(method) {
            currentSort = method;
            
            // Actualizar botones activos
            document.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');
            
            displayResults();
        }

        function toggleOnlyPopular() {
            showOnlyPopular = !showOnlyPopular;
            event.target.classList.toggle('active');
            filterResults();
        }

        function showError(message) {
            document.getElementById('error').textContent = message;
            document.getElementById('error').style.display = 'block';
            document.getElementById('loading').style.display = 'none';
        }
    </script>
